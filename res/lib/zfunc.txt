vector<int> Z(const string& S) {
	vector<int> z(S.size());
	int l = -1, r = -1;
	for(int i=1;i<S.size();i++) {
		z[i] = i >= r ? 0 : min(r - i, z[i - l]);
		while (i + z[i] < sz(S) && S[i + z[i]] == S[z[i]])
			z[i]++;
		if (i + z[i] > r)
			l = i, r = i + z[i];
	}
	return z;
}
