vector<string> duval(const string &s) {
   int n = sz(s), i = 0;
   vector<string> ret;
   while (i < n) {
      int j = i + 1, k = i;
      while (j < n && s[k] <= s[j])
         k = (s[k] < s[j] ? i : k + 1), j++;
      while (i <= k)
         ret.pb(s.substr(i, j - k)), i += j - k;
   }
   return ret;
}
